            CAST(DATEDIFF(MINUTE, APERTURA, PREPARACION) / 60 AS INTEGER) = 1,
            ' hora',
            ' horas'
        )
        || IIF(
            MOD(DATEDIFF(MINUTE, APERTURA, PREPARACION), 60) > 0,
            ' ' || MOD(DATEDIFF(MINUTE, APERTURA, PREPARACION), 60) || ' min.',
            ''
        )
    )
  ) AS TOTAL_ATENCION,


  FECHA_SALIDA,
  FECHA_REGRESO
FROM(
     SELECT
       CMD.IDCOMANDA,
       ORG.DESCRIPCION,
       SUC.NOMBRE AS SUCURSAL,
       CMD.MODULO AS CANAL,
       CMD.TOTAL AS IMPORTE,
       PER.PERSONA AS CLIENTE,
       COALESCE(TRIM(PER.TELEFONO), '?') AS TELEFONO,
       DRI.APELLIDO ||', '|| DRI.NOMBRE AS SERVICIO,
       CMD.NRO_COMANDA AS PEDIDO,
       CMD.APERTURA,
       COALESCE((SELECT x3.FECHA_HORA FROM SP_RECUPERA_TIEMPOS_COMANDA(CMD.idcomanda, 2) x3), CMD.APERTURA) AS COCINA,
       COALESCE((SELECT x4.FECHA_HORA FROM SP_RECUPERA_TIEMPOS_COMANDA(CMD.idcomanda, 3) x4), NULL) AS PREPARACION,
       COALESCE((SELECT x4.FECHA_HORA FROM SP_RECUPERA_TIEMPOS_COMANDA(CMD.idcomanda, 4) x4), NULL) AS ENTREGA,
       COALESCE(IIF(CMD.MODULO = 1, CMD.FECHA_SALIDA,NULL), NULL) AS FECHA_SALIDA,
       COALESCE(IIF(CMD.MODULO = 1, CMD.FECHA_REGRESO,NULL), NULL) AS FECHA_REGRESO
     FROM
       GG_COMANDA CMD
     INNER JOIN
       PERSONA PER
       ON (PER.IdPersona = CMD.IdPersona)
     INNER JOIN
       GG_MESERO DRI
       ON (DRI.IdMesero = CMD.IdMesero)
     INNER JOIN
       GG_TURNO TUR
       ON (TUR.IdTurno = CMD.IdTurno)
     INNER JOIN
       SUCURSAL SUC
       ON (SUC.IdSucursal = CMD.IdSucursal)
     INNER JOIN
       GG_ORIGEN ORG
       ON (ORG.IDORIGEN = CMD.IDORIGEN)
     WHERE
       TUR.Apertura BETWEEN :vDesde AND :vHasta
       AND SUC.IdFranquicia = 1
       AND CMD.IdSucursal = :vSucursal
       AND CMD.MODULO = 1
       AND CMD.Estado <> 3
     )
( IDPEDIDO,
  ORIGEN,
  SUCURSAL,
  CANAL,
  IMPORTE,
  CLIENTE,
  TELEFONO,
  SERVICIO,
  PEDIDO,
  APERTURA,
  COCINA,
  PREPARACION,
  ENTREGA,
  FECHA_SALIDA,
  FECHA_REGRESO)
WHERE
  EXTRACTHOUR(APERTURA) >= :vHora
ORDER BY
  EXTRACTDATE(APERTURA),
  CANAL
